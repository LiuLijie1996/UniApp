### 下拉刷新的两种方式

+ 通过页面配置

  ```json
  {
      "pages":[
          {
              "path":"pages/index/index",
              "style":{
                  "enablePullDownRefresh": true,//开启下拉样式
                  "navigationBarTitleText": "信息页"//页面标题,
                  "pullToRefresh":{//下拉刷新的样式
                      "color":"#007AFF"//图标的颜色
                  }
              }
          }
      ]
  }
  ```

+ 通过 `uni.startPullDownRefresh()` 方法来开启下拉刷新

  ```html
  <template>
  	<view class="">
  		<view class="msg">{{msg}}</view>
  
  		<button type="default" @click="pullDown">点击刷新</button>
  	</view>
  </template>
  
  <script>
  	export default {
  		data() {
  			return {
  				msg: "你好世界"
  			}
  		},
  
  		// 监听用户下拉动作，一般用于下拉刷新
  		onPullDownRefresh() {
  			setTimeout(() => {
  				// 修改msg数据
  				this.msg = 'hello world';
  				// 关闭下拉刷新
  				uni.stopPullDownRefresh();
  			}, 1000)
  		},
  
  		methods: {
  			pullDown() {
  				// 触发下拉刷新的api
  				uni.startPullDownRefresh()
  			}
  		}
  	}
  </script>
  ```



### 触底时添加数据到页面

+ 页面配置

  ```json
  {
      "pages":[
  		{
  			"path": "pages/list/list",
  			"style": {
  				"navigationBarTitleText": "信息页",
  				"onReachBottomDistance": 100//页面上拉触底事件触发时距页面底部距离，单位只支持px
  			}
  		}
      ]
  }
  ```

+ 添加数据到页面

  ```js
  export default {
  	data() {
  		return {
  			list: ['语文','数学','英语','地理','历史','生物','政治']
  		}
  	},
  
  	// 页面上拉触底时触发
  	onReachBottom() {
  		// 添加数据给list
  		this.list.push('化学','属性','内容','成功','编译','遍历','循环');
  	}
  }
  ```
